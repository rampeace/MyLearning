2. You have a Person object with an IsPremiumCustomer boolean. How would you change the background color of a Border to gold if the person is premium, otherwise keep it white?

    ✅ Tests use of DataTriggers or converters with styles.

    <Border Padding="10" Margin="10">
    <Border.Style>
        <Style TargetType="Border">
            <Setter Property="Background" Value="White" />
            <Style.Triggers>
                <DataTrigger Binding="{Binding IsPremiumCustomer}" Value="True">
                    <Setter Property="Background" Value="Gold"/>
                </DataTrigger>
            </Style.Triggers>
        </Style>
    </Border.Style>

    <StackPanel>
        <TextBlock Text="{Binding Name}" FontSize="16"/>
    </StackPanel>
</Border>

3. How do you show or hide a section (like a Grid or StackPanel) based on a ViewModel property, say IsVisibleSection (bool)? What are the ways to achieve it?

    ✅ Tests binding to Visibility with BooleanToVisibilityConverter or custom logic.

<Window.Resources>
    <BooleanToVisibilityConverter x:Key="BoolToVis"/>
</Window.Resources>

<StackPanel Visibility="{Binding IsVisibleSection, Converter={StaticResource BoolToVis}}">
    <!-- content here -->
</StackPanel>

Option 2:
public class BoolToHiddenConverter : IValueConverter
{
    public object Convert(object value, Type targetType, object parameter, CultureInfo culture)
    {
        bool isVisible = (bool)value;
        return isVisible ? Visibility.Visible : Visibility.Hidden;
    }

    public object ConvertBack(object value, Type targetType, object parameter, CultureInfo culture)
    {
        return (Visibility)value == Visibility.Visible;
    }
}

<local:BoolToHiddenConverter x:Key="BoolToHidden"/>
<StackPanel Visibility="{Binding IsVisibleSection, Converter={StaticResource BoolToHidden}}"/>

4. You have an Age property. If Age < 18, display text in red. If Age >= 18, display in green. How would you implement this purely in XAML?

    ✅ Tests DataTriggers, ValueConverters, or styles.

public class Person
{
    public int Age { get; set; }

    public bool IsMinor => Age < 18;
}

<TextBlock Text="{Binding Age}">
    <TextBlock.Style>
        <Style TargetType="TextBlock">
            <Setter Property="Foreground" Value="Green"/>
            <Style.Triggers>
                <DataTrigger Binding="{Binding IsMinor}" Value="True">
                    <Setter Property="Foreground" Value="Red"/>
                </DataTrigger>
            </Style.Triggers>
        </Style>
    </TextBlock.Style>
</TextBlock>

5. Suppose you have a ListBox showing a list of employees, and you want to show the selected employee's full details in a panel below it. How would you bind that?

    ✅ Tests understanding of SelectedItem binding and using DataTemplates for type-based rendering.

public class Employee
{
    public string Name { get; set; }
    public int Age { get; set; }
    public string Department { get; set; }
}

public class EmployeeViewModel
{
    public ObservableCollection<Employee> Employees { get; set; }
    public Employee SelectedEmployee { get; set; }

    public EmployeeViewModel()
    {
        Employees = new ObservableCollection<Employee>
        {
            new Employee { Name = "John", Age = 30, Department = "HR" },
            new Employee { Name = "Priya", Age = 25, Department = "Finance" }
        };
    }
}

<Window ...
        xmlns:local="clr-namespace:YourAppNamespace">
    <Window.DataContext>
        <local:EmployeeViewModel/>
    </Window.DataContext>

    <StackPanel>
        <!-- List of Employees -->
        <ListBox ItemsSource="{Binding Employees}"
                 SelectedItem="{Binding SelectedEmployee, Mode=TwoWay}"
                 DisplayMemberPath="Name" />

        <!-- Details Panel -->
        <ContentControl Content="{Binding SelectedEmployee}">
            <ContentControl.ContentTemplate>
                <DataTemplate>
                    <StackPanel Margin="10">
                        <TextBlock Text="{Binding Name}" FontWeight="Bold" />
                        <TextBlock Text="{Binding Age}" />
                        <TextBlock Text="{Binding Department}" />
                    </StackPanel>
                </DataTemplate>
            </ContentControl.ContentTemplate>
        </ContentControl>
    </StackPanel>
</Window>

6. How can you apply a specific Style to a TextBox only when it is focused, and a different one when not focused?

    ✅ Tests Triggers inside Style, and visual states.

    <Window.Resources>
    <Style TargetType="TextBox" x:Key="FocusAwareTextBoxStyle">
        <Setter Property="Background" Value="White"/>
        <Setter Property="BorderBrush" Value="Gray"/>
        <Setter Property="BorderThickness" Value="1"/>

        <Style.Triggers>
            <Trigger Property="IsFocused" Value="True">
                <Setter Property="Background" Value="LightYellow"/>
                <Setter Property="BorderBrush" Value="DodgerBlue"/>
                <Setter Property="FontWeight" Value="Bold"/>
            </Trigger>
        </Style.Triggers>
    </Style>
</Window.Resources>

<!-- Apply the style -->
<TextBox Style="{StaticResource FocusAwareTextBoxStyle}" 
         Width="200" Margin="20"/>


7. You want to highlight (bold or change background) the item in a ListBox if its IsManager property is true. How would you implement this?

    ✅ Tests DataTrigger inside ItemContainerStyle or ItemTemplate.

<Window.DataContext>
    <local:EmployeeViewModel/>
</Window.DataContext>

<ListBox ItemsSource="{Binding Employees}">
    <ListBox.ItemContainerStyle>
        <Style TargetType="ListBoxItem">
            <Setter Property="Background" Value="White"/>
            <Setter Property="FontWeight" Value="Normal"/>

            <Style.Triggers>
                <DataTrigger Binding="{Binding IsManager}" Value="True">
                    <Setter Property="Background" Value="LightGreen"/>
                    <Setter Property="FontWeight" Value="Bold"/>
                </DataTrigger>
            </Style.Triggers>
        </Style>
    </ListBox.ItemContainerStyle>

    <ListBox.ItemTemplate>
        <DataTemplate>
            <TextBlock Text="{Binding Name}" Padding="5"/>
        </DataTemplate>
    </ListBox.ItemTemplate>
</ListBox>
